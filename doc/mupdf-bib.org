(require 'bibtex-completion)
(require 'org-ref)
;; from https://github.com/enometh/org-media-note-ref/tree/madhu/contrib
(require 'org-media-note)
(require 'org-media-note-org-ref)
(require 'omn-transient)
(require 'org-pdf-open)

(setq org-bibtex-prefix "BIB_")
(setq org-bibtex-export-arbitrary-fields t)

(defvar $mupdf-bib-dir "~/cl/extern/cl-claw-mupdf/doc/")
(setq bibtex-completion-bibliography
  (list (file-name-concat $mupdf-bib-dir "mupdf-bib.bib")))
(setq bibtex-completion-notes-path
      (file-name-concat $mupdf-bib-dir "notes/"))

(setq org-startup-folded 'nofold)
(setq org-startup-truncated nil)
(setq org-descriptive-links nil)

;; org-bibtex-export-to-kill-ring
;; (bibtex-completion-clear-cache)
;; (bibtex-completion-init)
;; org-bibtex-entries
;; (org-bibtex-check)
(when nil
(defun bibtex-to-org ()
  "read org-bibtex-entries from current buffer"
  (bibtex-map-entries (lambda (key beg end)
			(org-bibtex-read))))
(with-temp-buffer
  (while org-bibtex-entries (org-bibtex-write))
  (write-region (point-min) (point-max) "/dev/shm/1"))
(with-temp-buffer
  (insert-file-contents "/dev/shm/1")
  (org-mode)
  (goto-char (point-min))
  (org-bibtex "/dev/shm/1.bib"))
)
#+end_src

  (org-mode)
  (goto-char (point-min))

				"mupdf-bib.bib")))
